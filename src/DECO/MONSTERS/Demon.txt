ACTOR SmoothDemon : Demon
{
	States
	{
	Spawn:
		SARG AB 10 fast A_Look
		Loop
	See:
		SSRG AABCDDEF 2 fast A_Chase
		Loop
	Melee:
		SSRG G 8 fast A_FaceTarget
		SSRG GGGHIII 1 fast A_FaceTarget
		SSRG J 1 fast
		SSRG K 1 fast A_SargAttack
		SSRG L 4 fast
		SSRG G 3 fast
		Goto See
	Pain:
		SSRG M 1 fast
		SSRG N 2 fast
		SSRG M 1 fast A_Pain
		Goto See
	Death.GibbyHaynes:
		TNT1 A 0
		TNT1 A 0 A_JumpIf(CallACS("StudioGibbly")==0, "Death")
		TNT1 A 0
		TNT1 A 0 A_JumpIfCloser(128, "GibDeath")
		TNT1 A 0
	Death:
		TNT1 a 0 A_JumpIfHealthLower(-30,"Gibdeath")
		SSRD A 3
		SSRD B 3 A_Scream
		SSRD C 3
		SSRD D 3 A_NoBlocking
		SSRD EFGH 3
		SSRD I -1
		Stop
	GibDeath:
		SARG o 3
		TNT1 AAAAAAAAAAAAAAA 0 A_CustomMissile ("Gibs1", 32, 0, random (0, 360), 2, random (0, 160))
		TNT1 AA 0 A_CustomMissile ("Gibs2", 32, 0, random (0, 360), 2, random (0, 160))
		TNT1 AAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAA 0 A_SpawnItem("XBlood2", 0, 40)
		SARG p 0 a_playsound ("misc/gibbed")
		SARG p 3 a_scream
		SARG q 3
		SARG r 3 a_noblocking
		SARG stuvw 3
		SARG x -1
		stop
	}
}
