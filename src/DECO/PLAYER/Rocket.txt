		TNT1 A 0 A_JumpIf(CallACS("64Launchertoggle")==1, "64spawn")

		Loop
	64Spawn:
		TNT1 A 0 A_JumpIf(CallACS("64Launchertoggle")==0, "spawn")
		RO64 z 2
		Loop
	Ready:
		TNT1 A 0 A_JumpIf(CallACS("64Launchertoggle")==1, "64ready")
		MISG A 1 A_WeaponReady
		Loop

		TNT1 A 0 A_JumpIf(CallACS("64Launchertoggle")==1, "64deselect")
		MISG A 1 A_Lower
		Loop

		TNT1 A 0 A_JumpIf(CallACS("64Launchertoggle")==1, "64select")
		MISG A 1 A_Raise
		Loop

		RO64 A 1 A_WeaponReady
		Loop

		Loop

		RO64 A 1 A_Raise

		PKRL FG 1
		PKRL A 0 A_ReFire
		Goto Ready
	RecoilFire:

		TNT1 A 0 A_JumpIf(CallACS("64Launchertoggle")==1, "64recoilfire")

		PKRL A 4 A_GunFlash
		PKRL B 2
		PKRL D 1 A_FireMissile

		PKRL G 1 a_setpitch(pitch+0.1)

		PKRL A 0 A_ReFire
		Goto Ready

		TNT1 A 0 A_JumpIf(CallACS("64Launchertoggle")==1, "64recoilfire")

		TNT1 A 0 A_WeaponReady(WRF_NoFire|WRF_NoSwitch)

		PKRL D 1 A_FireMissile
		TNT1 A 0 A_WeaponReady(WRF_NoFire|WRF_NoSwitch)

		TNT1 A 0 A_WeaponReady(WRF_NoFire|WRF_NoSwitch)

		TNT1 A 0 A_WeaponReady(WRF_NoFire|WRF_NoSwitch)
